let {EventEmitter} = require('events') - events - модуль

let ee = new EventEmitter();

ee.on('Log' - назва метода, (name) =>{
    console.log(`Log is working ${name}`) - тут прописуємо коллбек функцію
})

ee.emit('Log', 'Oleg') - викликати емітер який спрацює стільки разів скільки викличеш

ee.once('') - спрацьовує лише раз

ee.eventNames - повертає масив з назв івентів
